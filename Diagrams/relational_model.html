Music Streaming Service - Final Relational Model

  • User(Username, Password, First_Name, Last_Name, Email, Registration_Date)
  • Profile_Page(URL, Visibility, Username)
    • FK: Username ⊆ User(Username)
  • Regular_User(Username, Subscription_Type, Payment_Method, Renewal_Date)
    • FK: Username ⊆ User(Username)
  • Artist(Username, Stage_Name, Biography)
    • FK: Username ⊆ User(Username)
  • Album(Title, Primary_Artist_Username, Release_Date, Total_Duration, Description)
    • FK: Primary_Artist_Username ⊆ Artist(Username)
  • Album_Artist(Album_Title, Primary_Artist_Username, Contributing_Artist_Username, Role)
    • FK: (Album_Title, Primary_Artist_Username) ⊆ Album(Title, Primary_Artist_Username)
    • FK: Contributing_Artist_Username ⊆ Artist(Username)
  • Song(Title, Album_Title, Primary_Album_Artist_Username, File_Path, Duration, Lyrics, Track_Number)
    • FK: (Album_Title, Primary_Album_Artist_Username) ⊆ Album(Title, Primary_Artist_Username)
  • Song_Artist(Song_Title, Album_Title, Primary_Album_Artist_Username, Performing_Artist_Username, Role)
    • FK: (Song_Title, Album_Title, Primary_Album_Artist_Username) ⊆ Song(Title, Album_Title, Primary_Album_Artist_Username)
    • FK: Performing_Artist_Username ⊆ Artist(Username)
  • Genre(Name)
  • Album_Genre(Album_Title, Primary_Artist_Username, Genre_Name)
    • FK: (Album_Title, Primary_Artist_Username) ⊆ Album(Title, Primary_Artist_Username)
    • FK: Genre_Name ⊆ Genre(Name)
  • Song_Genre(Song_Title, Album_Title, Primary_Album_Artist_Username, Genre_Name)
    • FK: (Song_Title, Album_Title, Primary_Album_Artist_Username) ⊆ Song(Title, Album_Title, Primary_Album_Artist_Username)
    • FK: Genre_Name ⊆ Genre(Name)
  • Playlist(Title, Creator_Username, Creation_Date, Is_Public, Description)
    • FK: Creator_Username ⊆ User(Username)
  • Includes(Playlist_Title, Creator_Username, Song_Title, Album_Title, Primary_Album_Artist_Username, Added_Date)
    • FK: (Playlist_Title, Creator_Username) ⊆ Playlist(Title, Creator_Username)
    • FK: (Song_Title, Album_Title, Primary_Album_Artist_Username) ⊆ Song(Title, Album_Title, Primary_Album_Artist_Username)
  • Follows(Follower, Following)
    • FK: Follower ⊆ User(Username)
    • FK: Following ⊆ User(Username)
  • Saves_Playlist(Username, Playlist_Title, Creator_Username)
    • FK: Username ⊆ User(Username)
    • FK: (Playlist_Title, Creator_Username) ⊆ Playlist(Title, Creator_Username)
  • Saves_Album(Username, Album_Title, Primary_Artist_Username)
    • FK: Username ⊆ User(Username)
    • FK: (Album_Title, Primary_Artist_Username) ⊆ Album(Title, Primary_Artist_Username)
